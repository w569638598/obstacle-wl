##### 导出方法 export methods
```javascript


import x6 from './x6'

const components = [x6]

function install(Vue){
    components.forEach(comp => {
        Vue.component(comp.name, comp)
    })
}

if(typeof window !== "undefined" && window.Vue){
    install(window.Vue)
}

export { x6 }

export default { install }
```
